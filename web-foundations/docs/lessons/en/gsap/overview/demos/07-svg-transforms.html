// Basic SVG transforms with GSAP
gsap.to('.circle', {
  scale: 1.5,
  rotation: 360,
  transformOrigin: '50% 50%',  // Center of element
  duration: 2,
  ease: 'elastic.out(1, 0.3)',
  repeat: -1,       // Infinite loop
  yoyo: true        // Play forward then backward
});
